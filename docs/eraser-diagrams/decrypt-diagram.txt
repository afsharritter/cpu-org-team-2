title Decrypt File
autoNumber off


user [icon: monitor, label: "User"]
orch [icon: monitor, color: gray, label: "RSA Orchestrator"]
rsa [icon: gear, color: blue, label: "RSA Lib"]
fs [icon: folder, color: yellow, label: "File System"]

user > orch: Run RSA.s
orch > user: Prompt menu
user > orch: Select 3. Decrypt File
Keys [label: "2. Decrypt File"] {
  user <> orch: Prompt d & n
  user <> orch: Prompt Filename
  orch > orch: Extract file text
  loop [label: "for each encrypted character"] {
    orch > orch: Get ASCII Equivalent
    orch > rsa: decrypt(c, e, n)
    rsa > rsa: pow(c, e)
    rsa > rsa: mod(n)
    rsa > orch: return m

  }
  orch > user: Display m[]
  orch > fs: Save decrypted.txt 
}
